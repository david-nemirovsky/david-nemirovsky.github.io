Some Cool Plotly Plots

Data Import, Sampling, and Tidying

First, let’s get the restaurant data:

get_all_inspections = function(url) {
  
  all_inspections = vector("list", length = 0)
  
  loop_index = 1
  chunk_size = 50000
  DO_NEXT = TRUE
  
  while (DO_NEXT) {
    message("Getting data, page ", loop_index)
    
    all_inspections[[loop_index]] = 
      GET(url,
          query = list(`$order` = "zipcode",
                       `$limit` = chunk_size,
                       `$offset` = as.integer((loop_index - 1) * chunk_size)
                       )
          ) %>%
      content("text") %>%
      fromJSON() %>%
      as_tibble()
    
    DO_NEXT = dim(all_inspections[[loop_index]])[1] == chunk_size
    loop_index = loop_index + 1
  }
  
  all_inspections
  
}

url = "https://data.cityofnewyork.us/resource/43nn-pn8j.json"

nyc_inspections = 
  get_all_inspections(url) %>%
  bind_rows()

Now, let’s take a sample of 10,000 restaurant inspections in NYC:

set.seed(6)
sample_df = 
  nyc_inspections %>% 
  sample_n(10000)

Time for some data tidying:

sample_df = 
  sample_df %>% 
  mutate(
    borough = as.factor(boro),
    critical_flag = as.factor(critical_flag),
    grade = as.factor(grade),
    score = as.double(score),
    grade_date = as.Date(grade_date)) %>% 
    select(dba, borough, cuisine_description, critical_flag, score, grade, grade_date) %>% 
    filter(grade == c("A", "B", "C")) %>% 
    filter(borough != "0")

The resulting sample dataset is composed of 1649 observations.

Plot 1

Now, let’s plot the mean inspection scores for all types of restaurants in the 5 boroughs within our sample (note that higher scores result from more hazardous health violations):

sample_df %>% 
  group_by(borough, score) %>% 
  summarize(mean_score = mean(score)) %>% 
  plot_ly(x = ~borough, y = ~mean_score, type = "box", color = ~borough, colors = "viridis") %>% 
  layout(title = "Mean Inspection Scores for All Restauarnt Types", xaxis = list(title = "Borough"), yaxis = list(title = "Mean Score"))

Plot 2

This time, let’s make a plot to see the distribution of grades for Chinese food restaurants throughout the 5 boroughs from our sample:

sample_df %>% 
  filter(str_detect(cuisine_description, "[Cc][Hh][Ii][Nn][Ee][Ss][Ee]")) %>% 
  group_by(borough, grade) %>% 
  count(grade) %>% 
  plot_ly(x = ~grade, y = ~n, color = ~borough, type = "bar", colors = "viridis") %>% 
  layout(title = "Distribution of Chinese-Food Restaurant Inspection Grades", xaxis = list(title = "Grade"), yaxis = list(title = "Count"))
